203 inline MapDBReaderConfig read_db_map_reader_config_from_env(ILogger::SharedPtr logger)
204 {
205  MapDBReaderConfig config{};
206 
207  if (const char* map_storage_dir = std::getenv("PERSISTENT_STORAGE_ROOT_DIR"))
208  {
209  config.db_map_persistent_storage_root_dir = map_storage_dir;
210  }
211  else
212  {
213  logger->error("PERSISTENT_STORAGE_ROOT_DIR not set.");
214  }
215 
216  if (const char* map_catalog = std::getenv("DB_MAP_CATALOG"))
217  {
218  config.db_map_catalog = map_catalog;
219  }
220  else
221  {
222  logger->info("DB_MAP_CATALOG not set. Using default map catalog.");
223  }
224 
225  if (const char* map_endpoint = std::getenv("DB_MAP_ENDPOINT"))
226  {
227  config.db_map_endpoint = map_endpoint;
228  }
229  else
230  {
231  logger->info("DB_MAP_ENDPOINT not set up. Map endpoint set by default.");
232  }
233 
234  if (const char* map_env = std::getenv("DB_MAP_VERSION"))
235  {
236  std::istringstream iss{map_env};
237  std::uint64_t map_version = 0;
238  iss >> map_version;
239  config.db_requested_map_version = map_version;
240  }
241  else
242  {
243  logger->info("DB_MAP_VERSION not set up. Local map is used, no over the air update is possible.");
244  }
245 
246  return config;
247 }

255 inline std::shared_ptr<dbs_map::MapService> create_map_service(
256  const MapDBReaderConfig& config, ILogger::SharedPtr logger)
257 {
258  auto map_service_config = dbs_map::GetDefaultConfig();
259 
260  map_service_config.map_local_path_ =
261  (std::filesystem::path{config.db_map_persistent_storage_root_dir} / "hdmap").string();
262 
263  if (config.db_map_catalog)
264  {
265  map_service_config.catalog_ = config.db_map_catalog.value();
266  }
267 
268  if (config.db_map_endpoint)
269  {
270  if (*config.db_map_endpoint == "DBMC")
271  {
272  logger->info("DB_MAP_ENDPOINT = DBMC");
273  map_service_config.http_client_settings_ = dbs_map::download::GetDBClientSettings();
274  }
275  else if (*config.db_map_endpoint == "DB_LAN")
276  {
277  logger->info("DB_MAP_ENDPOINT = DB_LAN");
278  map_service_config.http_client_settings_ = dbs_map::download::GetDBClientForCISettings();
279  }
280  else
281  {
282  logger->error("DB_MAP_ENDPOINT is invalid. Map endpoint set by default.");
283  }
284  map_service_config.http_client_settings_.verbose_ = true;
285  }
286  return std::make_shared<dbs_map::MapService>(map_service_config);
287 }
